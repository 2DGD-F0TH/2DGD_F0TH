// A is an array of integers
int n = A.size();
do{
    bool swapped = false;
    for (i = 0; i < n; ++i) {
        if A[i-1] > A[i]{
            int tmp = A[i-1];
            A[i-1] = A[i];
            A[i] = tmp;
            swapped = true;
        }
    }
}while(!swapped)

// -------------------------------------

for (auto item: A){
    std::cout << item << std::endl;
}
