class AuthenticationMixin(){
    /*
     * This class has the authentication functions that allow
     * to check for authentication
     */
    function is_authenticated(user){
        /*
         * Function used to check for authentication
         */
        // ...
    }

    function connect(request){
        /*
         * This function wraps around the normal connection and performs an
         * authentication check first
         */
        if is_authenticated(request.user){
            // We're authenticated, do whatever should be done
            return superclass.connect(request);
        // Not authenticated, do something else
        return ForbiddenErrorResponse();
    }
}

    /*
     * This class represents a generic endpoint
     */
    function do_stuff(){

    }
    function connect(request){

         * A simple connection function for this endpoint

}
